class Solution {
    public int[] solution(int brown, int yellow) {
        int[] answer = new int[2];
        
        int tot = brown + yellow;
        for(int i=1;i<=Math.sqrt(tot);i++){
            if(tot%i==0){
                int j = tot/i;
                if(i*2+(j-2)*2==brown && i*j-brown==yellow){
                    answer[0] =j;//ê°€ë¡œ
                    answer[1] =i;//ì„¸ë¡œ
                }
            }            
        } 
        return answer;
    }
}
